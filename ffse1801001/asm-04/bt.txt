        function inputarr(bt) {
            var icon, arr, i
            arr = []
            i = 0
            icon = '1'
            while (icon == '1') {
                icon = prompt('Bài ' + bt, 'Nhập 1 để tiếp tục 0 để dừng lại');
                if (icon != '1') {
                    break;
                }
                iarr = eval(prompt('Nhập giá trị của mảng', ''))
                arr[i] = iarr
                i++
            }
            return arr
        }

        function bt1() {

        //     5.	Viết hàm giải phương trình bậc hai với  a==0 chuyển thành bậc nhất. 



        function bt5() {
            var x1, x2;
            var delta;
            var a, b, c;
            a = eval(prompt('Nhập a (a#0)', ''))
            b = eval(prompt('Nhập b', ''))
            c = eval(prompt('Nhập c', ''))
            if (a == 0) {
                ptbn(b, c)
            } else {
                delta = (b * b - 4 * a * c);
                if (delta == 0) {
                    x1 = x2 = -b / (2 * a);
                    alert("Phương trình có nghiệm kép: ");
                    alert('x:' + x1);
                }
                else if (delta < 0) {
                    alert("Phương trình vô nghiệm!");
                }
                else {
                    alert("Phương trình có hai nghiệm phân biệt: ");
                    x1 = (-b - Math.sqrt(delta)) / (2 * a);
                    x2 = (-b + Math.sqrt(delta)) / (2 * a);
                    alert("x1: " + x1);
                    alert("x2: " + x2);
                }
            }

        }
        function ptbn(a, b) {
            if (a == 0) {
                if (b == 0) {
                    alert("Phương trình có vô số nghiệm ");
                } else {
                    alert("Phương trình vô nghiệm ");
                }
            } else {
                alert("Phương trình có 1 nghiệm " + (-b / a))
            }
        }
